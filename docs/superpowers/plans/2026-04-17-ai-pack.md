# AI-pack Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Добавить в MailAi надстройку AI-pack: автоматическую классификацию писем (Важное / Неважно) через OpenRouter, NL-правила пользователя с drag-to-rule, опциональную серверную синхронизацию папок, retention-GC с тредовой связностью.

**Architecture:** Отдельный Swift-пакет `AI` с `AIProvider` протоколом и реализацией `OpenRouterClient`; `Classifier` и `RuleEngine` как акторы; фоновая `ClassificationQueue` в `AppShell`, работающая на метаданных из `Storage`. UI встраивается в каркасные слоты, зарезервированные в v1 (sidebar секция, прогресс-бар, sync-иконка, Settings). Никакой локальной эвристики — все классификации идут через AI.

**Tech Stack:** Swift 6, SwiftUI + AppKit (для sync-иконки), SwiftNIO (уже в транспорте), GRDB 7.x, Security.framework (Keychain), URLSession (для OpenRouter HTTP + SSE), Swift Testing.

**Reference spec:** [`docs/superpowers/specs/2026-04-17-ai-pack-design.md`](../specs/2026-04-17-ai-pack-design.md)

**Precondition:** MVP v1 завершён. Ниже — актуальный статус по `bd` (на 2026-04-18).

```swift
import GRDB
import Core
import Foundation

public actor RulesRepository {
    private let pool: DatabasePool

    public init(pool: DatabasePool) { self.pool = pool }

    public func upsert(_ rule: Rule) async throws {
        try await pool.write { db in
            try db.execute(sql: """
                INSERT INTO rules (id, text, intent, enabled, created_at, source)
                VALUES (?, ?, ?, ?, ?, ?)
                ON CONFLICT(id) DO UPDATE SET
                    text = excluded.text,
                    intent = excluded.intent,
                    enabled = excluded.enabled,
                    source = excluded.source
                """,
                arguments: [rule.id.uuidString, rule.text, rule.intent.rawValue,
                            rule.enabled, rule.createdAt, rule.source.rawValue])
        }
    }

    public func delete(id: UUID) async throws {
        try await pool.write { db in
            try db.execute(sql: "DELETE FROM rules WHERE id = ?", arguments: [id.uuidString])
        }
    }

    public func setEnabled(id: UUID, enabled: Bool) async throws {
        try await pool.write { db in
            try db.execute(sql: "UPDATE rules SET enabled = ? WHERE id = ?",
                           arguments: [enabled, id.uuidString])
        }
    }

    public func all() async throws -> [Rule] {
        try await pool.read { db in try Self.decodeRows(db.fetchAllRules()) }
    }

    public func active() async throws -> [Rule] {
        try await pool.read { db in try Self.decodeRows(db.fetchActiveRules()) }
    }

    private static func decodeRows(_ rows: [Row]) throws -> [Rule] {
        try rows.map { row in
            guard let id = UUID(uuidString: row["id"]),
                  let intent = Rule.Intent(rawValue: row["intent"]),
                  let source = Rule.Source(rawValue: row["source"]) else {
                throw DecodingError.dataCorrupted(.init(codingPath: [], debugDescription: "rule row"))
            }
```

```swift
import Foundation

public enum SnippetExtractor {
    public static func extract(body: String, contentType: String, length: Int = 150) -> String {
        let stripped = stripHTMLIfNeeded(body, contentType: contentType)
        let noQuoted = stripQuotedReply(stripped)
        let noSig = stripSignature(noQuoted)
        let normalized = noSig.replacingOccurrences(of: "\\s+", with: " ", options: .regularExpression)
            .trimmingCharacters(in: .whitespacesAndNewlines)
        if normalized.count >= length {
            return String(normalized.prefix(length))
        }
        return normalized.padding(toLength: length, withPad: " ", startingAt: 0)
    }

    private static func stripHTMLIfNeeded(_ body: String, contentType: String) -> String {
        guard contentType.lowercased().contains("html") else { return body }
        return body.replacingOccurrences(of: "<[^>]+>", with: " ", options: .regularExpression)
            .replacingOccurrences(of: "&nbsp;", with: " ")
            .replacingOccurrences(of: "&amp;", with: "&")
            .replacingOccurrences(of: "&lt;", with: "<")
            .replacingOccurrences(of: "&gt;", with: ">")
    }

    private static func stripQuotedReply(_ body: String) -> String {
        body.split(separator: "\n", omittingEmptySubsequences: false)
            .prefix(while: { !$0.trimmingCharacters(in: .whitespaces).hasPrefix(">") &&
                              !$0.hasPrefix("On ") })
            .joined(separator: "\n")
    }

    private static func stripSignature(_ body: String) -> String {
        if let range = body.range(of: "\n-- \n") {
            return String(body[..<range.lowerBound])
        }
        return body
    }
}
```

```swift
import Core
import Foundation

public struct OpenRouterClient: AIProvider, Sendable {
    public let apiKey: String
    public let model: String
    public let session: URLSession

    public init(apiKey: String, model: String, session: URLSession = .shared) {
        self.apiKey = apiKey
        self.model = model
        self.session = session
    }

    public func complete(system: String, user: String, streaming: Bool)
        -> AsyncThrowingStream<String, Error>
    {
        AsyncThrowingStream { continuation in
            let task = Task {
                do {
                    var req = URLRequest(url: URL(string: "https://openrouter.ai/api/v1/chat/completions")!)
                    req.httpMethod = "POST"
                    req.setValue("Bearer \(apiKey)", forHTTPHeaderField: "Authorization")
                    req.setValue("application/json", forHTTPHeaderField: "Content-Type")
                    req.setValue("https://mailai.app", forHTTPHeaderField: "HTTP-Referer")
                    req.setValue("MailAi", forHTTPHeaderField: "X-Title")
                    let body = ChatRequest(
                        model: model,
                        messages: [.init(role: "system", content: system),
                                   .init(role: "user", content: user)],
                        stream: streaming,
                        temperature: 0.2,
                        maxTokens: 200
                    )
                    req.httpBody = try JSONEncoder().encode(body)

                    if streaming {
                        try await streamResponse(req, continuation: continuation)
                    } else {
                        try await nonStreamResponse(req, continuation: continuation)
                    }
                    continuation.finish()
                } catch {
                    continuation.finish(throwing: error)
                }
            }
            continuation.onTermination = { _ in task.cancel() }
        }
    }

    private func streamResponse(_ req: URLRequest,
                                continuation: AsyncThrowingStream<String, Error>.Continuation) async throws
    {
        let (bytes, response) = try await session.bytes(for: req)
        try validate(response)
        for try await line in bytes.lines {
            guard line.hasPrefix("data: ") else { continue }
            let payload = String(line.dropFirst(6))
            if payload == "[DONE]" { break }
            guard let data = payload.data(using: .utf8),
                  let delta = try? JSONDecoder().decode(StreamDelta.self, from: data),
                  let content = delta.choices.first?.delta.content else { continue }
            continuation.yield(content)
        }
    }

    private func nonStreamResponse(_ req: URLRequest,
                                   continuation: AsyncThrowingStream<String, Error>.Continuation) async throws
    {
        let (data, response) = try await session.data(for: req)
        try validate(response)
        let decoded = try JSONDecoder().decode(NonStreamResponse.self, from: data)
        if let content = decoded.choices.first?.message.content { continuation.yield(content) }
    }

    private func validate(_ response: URLResponse) throws {
        guard let http = response as? HTTPURLResponse else { throw OpenRouterError.invalidResponse }
        switch http.statusCode {
        case 200..<300: return
        case 401: throw OpenRouterError.unauthorized
        case 402: throw OpenRouterError.paymentRequired
        case 429: throw OpenRouterError.rateLimited
        case 500..<600: throw OpenRouterError.serverError(http.statusCode)
        default: throw OpenRouterError.httpError(http.statusCode)
        }
    }
}

public enum OpenRouterError: Error, Equatable {
    case invalidResponse
    case unauthorized
    case paymentRequired
    case rateLimited
    case serverError(Int)
    case httpError(Int)
}
```

```swift
import Foundation

public enum ClassifierError: Error, Equatable {
    case malformedJSON
    case emptyResponse
}

public actor Classifier {
    private let provider: any AIProvider
    private let model: String

    public init(provider: any AIProvider, model: String) {
        self.provider = provider
        self.model = model
    }

    public func classify(input: ClassificationInput) async throws -> ClassificationResult {
        let started = Date()
        let prompt = ClassifyV1.build(input: input)
        var buffer = ""
        for try await chunk in provider.complete(system: prompt.system, user: prompt.user, streaming: false) {
            buffer += chunk
        }
        guard !buffer.isEmpty else { throw ClassifierError.emptyResponse }
        let parsed = try parseJSON(buffer)
        let durationMs = Int(Date().timeIntervalSince(started) * 1000)
        return ClassificationResult(
            importance: parsed.importance == "important" ? .important : .unimportant,
            confidence: parsed.confidence,
            matchedRule: nil,
            reasoning: parsed.reasoning,
            tokensIn: prompt.system.count / 4 + prompt.user.count / 4,  // грубая оценка
            tokensOut: buffer.count / 4,
            durationMs: durationMs
        )
    }

    private struct Parsed: Decodable {
        let importance: String
        let confidence: Double
        let reasoning: String
    }

    private func parseJSON(_ text: String) throws -> Parsed {
        let trimmed = extractJSON(text)
        guard let data = trimmed.data(using: .utf8),
              let parsed = try? JSONDecoder().decode(Parsed.self, from: data) else {
            throw ClassifierError.malformedJSON
        }
        return parsed
    }

    private func extractJSON(_ text: String) -> String {
        guard let start = text.firstIndex(of: "{"),
              let end = text.lastIndex(of: "}") else { return text }
        return String(text[start...end])
    }
}
```

```swift
import Core
import Foundation

public actor ClassificationQueue {
    public struct Snapshot: Sendable, Equatable {
        public let total: Int
        public let pending: Int
        public let failed: Int
        public let inFlight: Int
    }

    private let batchSize: Int
    private let maxParallel: Int
    private var pending: [UUID] = []
    private var inFlight: Set<UUID> = []
    private var failed: Set<UUID> = []
    private var total: Int = 0
    private var continuations: [UUID: AsyncStream<Snapshot>.Continuation] = [:]

    public init(batchSize: Int = 10, maxParallel: Int = 3) {
        self.batchSize = batchSize
        self.maxParallel = maxParallel
    }

    public func enqueue(_ ids: [UUID]) {
        pending.append(contentsOf: ids)
        total += ids.count
        broadcast()
    }

    public func snapshot() -> Snapshot {
        Snapshot(total: total, pending: pending.count, failed: failed.count, inFlight: inFlight.count)
    }

    public func observe() -> AsyncStream<Snapshot> {
        AsyncStream { continuation in
            let id = UUID()
            continuations[id] = continuation
            continuation.yield(snapshot())
            continuation.onTermination = { [weak self] _ in
                Task { await self?.removeContinuation(id: id) }
            }
        }
    }

    private func removeContinuation(id: UUID) {
        continuations.removeValue(forKey: id)
    }

    public func processAll(_ worker: @Sendable (UUID) async throws -> Void) async {
        await withTaskGroup(of: Void.self) { group in
            var active = 0
            while !pending.isEmpty || active > 0 {
                while active < maxParallel, let id = pending.first {
                    pending.removeFirst()
                    inFlight.insert(id)
                    broadcast()
                    active += 1
                    group.addTask { [weak self] in
                        do {
                            try await worker(id)
                            await self?.markDone(id: id)
                        } catch {
                            await self?.markFailed(id: id)
                        }
                    }
                }
                _ = await group.next()
                active -= 1
            }
        }
    }

    private func markDone(id: UUID) {
        inFlight.remove(id)
        broadcast()
    }

    private func markFailed(id: UUID) {
        inFlight.remove(id)
        failed.insert(id)
        broadcast()
    }

    private func broadcast() {
        let s = snapshot()
        for c in continuations.values { c.yield(s) }
    }
}
```

```swift
import Foundation

public actor UndoStack {
    public enum Action: Sendable, Equatable {
        case move(ids: [UUID], from: String, to: String)
    }

    private var items: [Action] = []
    private let capacity: Int

    public init(capacity: Int = 20) { self.capacity = capacity }

    public func push(_ action: Action) {
        items.append(action)
        if items.count > capacity { items.removeFirst(items.count - capacity) }
    }

    public func pop() -> Action? {
        items.popLast()
    }

    public func snapshot() -> [Action] { items }
}
```

```swift
import Core
import Storage
import MailTransport
import Foundation

public actor ServerFolderSync {
    public static let importantPath = "Отфильтрованные/Важное"
    public static let unimportantPath = "Отфильтрованные/Неважно"

    private let store: MetadataStore
    private let transport: any MailTransport
    private let undo: UndoStack
    private let accountId: UUID

    public init(store: MetadataStore, transport: any MailTransport, undo: UndoStack, accountId: UUID) {
        self.store = store
        self.transport = transport
        self.undo = undo
        self.accountId = accountId
    }

    public func migrateOn() async throws {
        try await transport.createMailbox(path: Self.importantPath)
        try await transport.createMailbox(path: Self.unimportantPath)

        let important = try await store.messageIds(accountId: accountId, importance: .important)
        let unimportant = try await store.messageIds(accountId: accountId, importance: .unimportant)

        try await moveBatched(important, to: Self.importantPath)
        try await moveBatched(unimportant, to: Self.unimportantPath)
    }

    public func syncSingle(messageId: UUID, to importance: Importance) async throws {
        let path = importance == .important ? Self.importantPath : Self.unimportantPath
        try await transport.move(messageIds: [messageId], to: path)
        await undo.push(.move(ids: [messageId], from: "INBOX", to: path))
    }

    public func migrateOff(revertToInbox: Bool) async throws {
        guard revertToInbox else { return }
        let all = try await store.messageIds(accountId: accountId, importance: .important)
                + store.messageIds(accountId: accountId, importance: .unimportant)
        try await moveBatched(all, to: "INBOX")
    }

    private func moveBatched(_ ids: [UUID], to path: String) async throws {
        for chunk in ids.chunked(by: 50) {
            try await transport.move(messageIds: chunk, to: path)
            await undo.push(.move(ids: chunk, from: "INBOX", to: path))
            try await Task.sleep(nanoseconds: 100_000_000)  // 100 ms между батчами
        }
    }
}

extension Array {
    func chunked(by size: Int) -> [[Element]] {
        stride(from: 0, to: count, by: size).map { Array(self[$0..<Swift.min($0 + size, count)]) }
    }
}
```

```swift
    public init(defaults: any UserDefaultsLike, gc: RetentionGC, onRun: @escaping @Sendable () -> Void = {}) {
        self.defaults = defaults
        self.gc = gc
        self.onRun = onRun
    }

    public func runIfDue(now: Date = .init()) async {
        let last = Date(timeIntervalSince1970: defaults.double(forKey: Self.key))
        guard now.timeIntervalSince(last) >= 24 * 3600 else { return }
        try? await gc.run(now: now)
        defaults.set(now.timeIntervalSince1970, forKey: Self.key)
        onRun()
    }
}

public protocol UserDefaultsLike: Sendable {
    func double(forKey: String) -> Double
    func set(_ value: Double, forKey: String)
}

extension UserDefaults: UserDefaultsLike {}
```
